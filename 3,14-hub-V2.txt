-- H√†m thay ƒë·ªïi t·ªëc ƒë·ªô ƒëi b·ªô
local function setWalkSpeed(speed)
    if LocalPlayer and LocalPlayer.Character then
        local Humanoid = LocalPlayer.Character:FindFirstChildOfClass("Humanoid")
        if Humanoid then
            Humanoid.WalkSpeed = speed
        end
    end
end

-- TƒÉng t·ªëc khi di chuy·ªÉn tr√™n m·∫∑t ƒë·∫•t
local function boostSpeedOnGround()
    RunService.RenderStepped:Connect(function()
        if LocalPlayer and LocalPlayer.Character then
            local Humanoid = LocalPlayer.Character:FindFirstChildOfClass("Humanoid")
            local RootPart = LocalPlayer.Character:FindFirstChild("HumanoidRootPart")

            if RootPart and Humanoid then
                -- Ki·ªÉm tra tr·∫°ng th√°i c·ªßa nh√¢n v·∫≠t khi di chuy·ªÉn tr√™n m·∫∑t ƒë·∫•t
                if Humanoid.MoveDirection.Magnitude > 0 and Humanoid:GetState() == Enum.HumanoidStateType.Physics then
                    -- Nh√¢n v·∫≠t ƒëang di chuy·ªÉn tr√™n m·∫∑t ƒë·∫•t, tƒÉng t·ªëc di chuy·ªÉn
                    RootPart.Velocity = Humanoid.MoveDirection * currentSpeed
                end
            end
        end
    end)
end

-- TƒÉng t·ªëc khi nh·∫£y v√† di chuy·ªÉn trong kh√¥ng kh√≠

    RunService.RenderStepped:Connect(function()
        if LocalPlayer and LocalPlayer.Character then
            local Humanoid = LocalPlayer.Character:FindFirstChildOfClass("Humanoid")
            local RootPart = LocalPlayer.Character:FindFirstChild("HumanoidRootPart")

            if RootPart and Humanoid then
                -- Ki·ªÉm tra tr·∫°ng th√°i c·ªßa nh√¢n v·∫≠t khi nh·∫£y
